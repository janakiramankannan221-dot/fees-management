1. Firestore Security Rules

Copy and paste these rules into your Firebase Console > Cloud Firestore > Rules tab to allow the app to work correctly.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone to read/write shared settings like notices
    match /settings/{document} {
      allow read, write: if true;
    }
    
    // Allow anyone to read all students (for the search/login flow)
    match /students/{studentId} {
      allow read: if true;
      // Writing to student records is restricted to Admins (Auth required)
      allow write: if request.auth != null;
    }
    
    // Payments subcollection
    match /events/{eventId}/payments/{regNo} {
      // Students can read their own payment status and create "Cash Pending" requests
      allow read: if true;
      allow create: if true;
      // Updating existing payments requires Admin Auth
      allow update, delete: if request.auth != null;
    }
    
    // Events collection
    match /events/{eventId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
